TITLE 32 BITLIK IKI SAYININ TOPLANMASI 

STACKSG SEGMENT PARA STACK 'STACK'

	DW 15 DUP(?)
STACKSG ENDS

DATASG SEGMENT PARA 'DATA'

	DD ASAYISI 12345678H
	DD BSAYISI 0ABCDEF0H
	DQ TOPLAM 0

DATASG ENDS

CODESG SEGMENT PARA 'CODE'
	ASSUME CS:CODESG,SS:STACKSG,DS:DATASG

BASLA PROC FAR
	
	PUSH DS
	XOR AX,AX
	PUSH AX

	MOV AX,DATASG
	MOV DS,AX

	LEA BX,TOPLAM
	LEA DI,ASAYISI
	LEA SI,BSAYISI

	MOV AX,[SI]
	MOV DX,[DI]
	ADD AX,DX
	MOV [BX],AX

	MOV AX,[SI+2]
	MOV DX,[DI+2]
	ADC AX,DX
	MOV [BX+2],AX
	ADC WORD PTR [BX+4],0	
	
	


RETF
BASLA ENDP
CODESG ENDS
END BASLA
