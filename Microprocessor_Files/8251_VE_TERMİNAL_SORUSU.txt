1- 8251 resetle									208 H adresine yerleşir.
2- mod değeri gönder							C/D ucu A1 de 
3- control değeri gönder
4- r x rdy = 1 olana kadar bekle
5- okuma yap
6- Ascıı değerini bir arttır
7- txrdy = 1 olana kadar bekle 
8- yazma yap

MOV DX,20AH
MOV AL,40H		; RESETLEMEK İÇİN VERİLEN DEĞER.
OUT DX,AL		; RESETLEDİM.

MOV AL,01001101B ; BURADA MOD YAZMACINI AYARLADIM.
OUT DX,AL

MOV AL,00010101B ; BURADA KONTROL DEĞERİMİ AYARLADIM.
OUT DX,AL 		 

EN_BAS:
;R x RDY OLANA KADAR BEKLEMELİYİM
MOV DX,20AH		; RxRDY UÇLARINI KONTROL ETMEK İÇİN C/D DEĞERİ 1 OLMALI O YÜZDEN 20AH DAN OKUMA YAPILIR.
BEKLE_R:
IN AL,DX
TEST AL,02H		; TEST İŞLEMİ AND YAPAR DEMEK Kİ 1 GELİRSE DEVAM ZERO GELİRSE RxRDY = 0 DEMEKTİR. BEKLEMEYE DEVAM ET. 
JZ BEKLE_R

;ARTIK RxRDY=1 İSE OKUMAYI YAPABİLİRİM.
;OKUMAYI C/D UCU 0 İKEN YAPABİLİRİM ARTIK 208H ADRESİNİ KULLANACAĞIM.

MOV DX,208H
IN AL,DX		;OKUMA YAPILDI
MOV BL,AL 		;DEĞERİ ARTIRMAK ÜZERE BL REGISTERINA ALDIM.
INC BL

;İŞLEMLER TAMAM TxRDY=1 OLANA KADAR BEKLEMELİYİM.
MOV DX,20AH		;BURADA TxRDY UCUNUN HAZIR OLUP OLMADIĞINI KONTROL DEĞERİNDEN OKUYACAĞIM C/D UCUNUN 1 OLMASI GEREKİR.
BEKLE_T:
IN AL,DX
TEST AL,01H		;BURADA TxRDY CONTROL WORDÜN A0 BİTİNDE O YÜZDEN 01 H İLE TEST EDİLDİ.
JZ BEKLE_T

; TxRDY=1 İSE YAZMA İŞLEMİNİ ARTIK YAPABİLİRİM.

; YAZMA İŞLEMİ İÇİN C/D UCUNUN 0 OLMASI LAZIM BU SEBEPLE 208H DEĞERİ KULLANILACAK.

MOV DX,208H 
MOV AL,BL 		;ARTIRDIĞIM DEĞERİ ALDIM.
OUT DX,AL  		;ARTIRDIĞIM DEĞERİ TERMİNALE BASTIM.
JMP EN_BAS
